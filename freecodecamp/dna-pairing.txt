function pairElement(str) {
  let dnaArray = [];
  for (let i = 0; i < str.length; i++) {
    if (str[i] == 'A') {
      dnaArray.push(['A', 'T'])
    } else if (str[i] == 'C') {
      dnaArray.push(['C', 'G'])
    } else if (str[i] == 'G') {
      dnaArray.push(['G', 'C'])
    } else if (str[i] == 'T') {
      dnaArray.push(['T', 'A'])
    }
  }
  return dnaArray
}

pairElement("GCG");
